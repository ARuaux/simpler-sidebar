Example of usage

$( document ).ready( function() {
  var $rightSidebar = $( "#sidebar-right" ),
    $leftSidebar = $( "#sidebar-left" ),
    callopen = function() {
      alert( "callback called after the sidebar's been opened" );
    },
    callclose = function() {
      alert( "callback called after the sidebar's been closed" );
    },
    callboth = function() {
      alert( "callback called after both animations" );
    },
    onopen = function() {
      alert( "function called while the sidebar is opening" );
    },
    onclose = function() {
      alert( "function called while the sidebar is closing" );
    },
    onboth = function() {
      alert( "function called while both animations are triggered" );
    },

    optSidebarLeft = {
      align: "left",
      selectors: {
        trigger: "#toggle-sidebar-right",
        quitter: ".close-sb"
      },
      animation: {
        easing: "easeOutQuint"
      }
    },
    optSidebarRight = {
      top: 56,
      selectors: {
        trigger: "#toggle-sidebar-left",
        quitter: ".close-sb"
      },
      sidebar: {
        width: 500
      },
      animation: {
        easing: "easeOutQuint"
      },
      mask: {
        display: false
      },
      events: {
        on: {
          animation: {
            open: onopen,
            close: onclose,
            both: onboth
          }
        },
        callbacks: {
          animation: {
            open: callopen,
            close: callclose,
            both: callboth,
            freezePage: false
          }
        }
      };

  $rightSidebar.simplerSidebar( optSidebarRight );
  $leftSidebar.simplerSidebar( optSidebarLeft );
