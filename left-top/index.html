Example of usage

$( "#sidebar-left" ).simplerSidebar( {
  top: 56,
  align: "left",
  selectors: {
    trigger: "#sidebar-main-trigger",
    quitter: ".quitter"
  },
  animation: {
    easing: "easeOutQuint"
  }
} );
